\subsection{定理}
\begin{frame}
    \begin{itemize}
        \item 定理按顺序连续编号。
    \end{itemize}
    \begin{theorem}[示例定理]
        设离散随机变量 $X$ 的取值空间为 $\mathcal{X}$，概率分布为 $p:{\mathcal{X}}\to[0,1]$，则其信息熵为：
        \begin{equation}
            \mathrm{H}(X) := -\sum_{x\in\mathcal{X}} p(x)\log p(x) = \mathbb{E}[-\log p(X)]
        \end{equation}
    \end{theorem}
\end{frame}

        \item 定义的编号以所在部分的节号为前缀。
